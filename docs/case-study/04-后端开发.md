#### AI 辅助场景 1：实现借用申请

```
👤 用户:
请帮我实现借用申请的业务逻辑，需要：
1. 检查设备状态是否为"空闲"
2. 检查设备是否已有待处理的借用记录
3. 创建借用记录，状态为"待审批"
4. 发送通知给审批人

> (输入自动发送到 AI)

🤖 AI Assistant:
我来帮你实现借用申请的业务逻辑。基于当前的 Service 层，我会添加 borrow() 方法：

```java
// BorrowRecordServiceImpl.java
package com.company.equipment.module.borrow.service.impl;

@Service
@RequiredArgsConstructor
public class BorrowRecordServiceImpl implements BorrowRecordService {
    
    private final BorrowRecordMapper borrowRecordMapper;
    private final EquipmentMapper equipmentMapper;
    private final SysUserMapper userMapper;
    private final MessageService messageService;
    
    @Override
    @Transactional(rollbackFor = Exception.class)
    public Long createBorrow(BorrowRecordCreateDTO dto, Long userId) {
        // 1. 检查设备是否存在
        Equipment equipment = equipmentMapper.selectById(dto.getEquipmentId());
        if (equipment == null) {
            throw new BusinessException("设备不存在");
        }
        
        // 2. 检查设备状态是否为"空闲"或"闲置"
        if (!Arrays.asList(0, 4).contains(equipment.getStatus())) {
            throw new BusinessException("设备当前不可借用");
        }
        
        // 3. 检查是否有待处理的借用记录
        Long pendingCount = borrowRecordMapper.countPendingByEquipmentId(
            dto.getEquipmentId()
        );
        if (pendingCount > 0) {
            throw new BusinessException("设备有待处理的借用申请");
        }
        
        // 4. 检查借用时间是否合理
        if (dto.getExpectReturn().isBefore(LocalDateTime.now())) {
            throw new BusinessException("预计归还时间不能早于当前时间");
        }
        
        // 5. 创建借用记录
        BorrowRecord record = new BorrowRecord();
        BeanUtils.copyProperties(dto, record);
        record.setUserId(userId);
        record.setBorrowDate(LocalDateTime.now());
        record.setStatus(BorrowStatus.PENDING.getCode());
        borrowRecordMapper.insert(record);
        
        // 6. 发送通知给审批人
        SysUser applicant = userMapper.selectById(userId);
        SysUser approver = findApprover(equipment);
        messageService.sendNotice(approver.getId(), 
            String.format("%s 申请借用设备 %s", 
                applicant.getRealName(), 
                equipment.getName()
            )
        );
        
        return record.getId();
    }
}
```

// 添加到 BorrowRecordServiceImpl.java
@Override
@Transactional(rollbackFor = Exception.class)
public void approve(Long recordId, Long approverId, boolean approved, String remark) {
    BorrowRecord record = borrowRecordMapper.selectById(recordId);
    if (record == null) {
        throw new BusinessException("借用记录不存在");
    }
    
    if (record.getStatus() != BorrowStatus.PENDING.getCode()) {
        throw new BusinessException("该记录不在待审批状态");
    }
    
    // 更新审批信息
    record.setApproverId(approverId);
    record.setApproveTime(LocalDateTime.now());
    record.setRemark(remark);
    
    if (approved) {
        // 审批通过
        record.setStatus(BorrowStatus.APPROVED.getCode());
        
        // 更新设备状态为"使用中"
        Equipment equipment = equipmentMapper.selectById(record.getEquipmentId());
        equipment.setStatus(EquipmentStatus.IN_USE.getCode());
        equipmentMapper.updateById(equipment);
        
        // 发送通知
        messageService.sendNotice(record.getUserId(),
            String.format("您的借用申请已通过，请在3天内领取设备 %s", 
                equipment.getName()));
    } else {
        // 审批拒绝
        record.setStatus(BorrowStatus.REJECTED.getCode());
        
        // 发送通知
        messageService.sendNotice(record.getUserId(),
            String.format("您的借用申请已被拒绝，原因：%s", remark));
    }
    
    borrowRecordMapper.updateById(record);
}

// 添加到 BorrowRecordServiceImpl.java

@Override
@Transactional(rollbackFor = Exception.class)
public void returnEquipment(Long recordId, Long userId, String remark) {
    BorrowRecord record = borrowRecordMapper.selectById(recordId);
    if (record == null) {
        throw new BusinessException("借用记录不存在");
    }
    
    if (record.getUserId().equals(userId)) {
        throw new BusinessException("非借用人不能操作归还");
    }
    
    if (record.getStatus() != BorrowStatus.IN_USE.getCode()) {
        throw new BusinessException("设备不在使用中状态");
    }
    
    // 更新借用记录
    record.setActualReturn(LocalDateTime.now());
    record.setStatus(BorrowStatus.RETURNED.getCode());
    
    // 检查是否超期
    if (record.getActualReturn().isAfter(record.getExpectReturn())) {
        record.setStatus(BorrowStatus.OVERDUE.getCode());
        // 发送超期提醒
        messageService.sendNotice(record.getUserId(),
            String.format("您借用的设备 %s 已超期，请尽快归还", 
                equipmentMapper.selectById(record.getEquipmentId()).getName()));
    }
    
    // 更新设备状态为空闲
    Equipment equipment = equipmentMapper.selectById(record.getEquipmentId());
    equipment.setStatus(EquipmentStatus.IDLE.getCode());
    equipmentMapper.updateById(equipment);
    
    // 更新借用记录备注
    if (StringUtils.isNotBlank(remark)) {
        record.setRemark(remark);
    }
    
    borrowRecordMapper.updateById(record);
}

## lnngfar backend 功能需求清单

通过这个案例，**lnngfar backend** 需要具备以下能力：

### 核心能力

| 能力 | 说明 | 实现方式 |
|------|------|----------|
| **模块创建** | 创建标准化的模块目录结构 | Template Engine |
| **AI 对话** | 实时 AI 对话辅助开发 | AI Runtime |
| **上下文感知** | 自动加载项目上下文到 AI | Context Management |
| **代码补全** | 基于上下文的代码补全 | AI Inference |
| **代码生成** | 根据描述生成代码片段 | AI Generation |

### AI 能力矩阵

| 能力 | 描述 | 优先级 |
|------|------|--------|
| **需求理解** | 理解自然语言需求 | P0 |
| **代码生成** | 生成符合规范的代码 | P0 |
| **上下文管理** | 维护长期开发上下文 | P0 |
| **规范遵循** | 遵循项目编码规范 | P1 |
| **多轮对话** | 支持多轮需求迭代 | P1 |
| **代码解释** | 解释现有代码逻辑 | P2 |

### 后端开发规范

| 规范 | 说明 |
|------|------|
| **分层规范** | Controller → Service → Mapper |
| **DTO 规范** | Create/Update/Query/VO 四类 DTO |
| **异常处理** | 统一异常处理 + 业务异常 |
| **日志规范** | 使用 Slf4j + Logback |
| **事务规范** | Service 层加事务注解 |
