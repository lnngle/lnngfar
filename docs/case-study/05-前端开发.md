<script setup lang="ts">
import { ref, reactive, onMounted } from 'vue'
import { ElMessage, ElMessageBox } from 'element-plus'
import { getEquipmentPage, deleteEquipment, exportEquipment } from '@/api/equipment'
import EquipmentForm from './components/EquipmentForm.vue'
import type { Equipment, EquipmentQuery } from '@/types/equipment'

// 状态
const loading = ref(false)
const tableData = ref<Equipment[]>([])
const total = ref(0)
const selectedIds = ref<number[]>([])
const formVisible = ref(false)
const currentRow = ref<Equipment | null>(null)
const categoryOptions = ref<{id: number, name: string}[]>([])

// 查询表单
const queryForm = reactive<EquipmentQuery>({
  name: '',
  categoryId: undefined,
  status: undefined,
  pageNum: 1,
  pageSize: 10
})

// 加载数据
const loadData = async () => {
  loading.value = true
  try {
    const res = await getEquipmentPage(queryForm)
    tableData.value = res.data.records
    total.value = res.data.total
  } finally {
    loading.value = false
  }
}

// 查询
const handleQuery = () => {
  queryForm.pageNum = 1
  loadData()
}

// 重置
const handleReset = () => {
  queryForm.name = ''
  queryForm.categoryId = undefined
  queryForm.status = undefined
  queryForm.pageNum = 1
  loadData()
}

// 新增
const handleAdd = () => {
  currentRow.value = null
  formVisible.value = true
}

// 编辑
const handleEdit = (row: Equipment) => {
  currentRow.value = { ...row }
  formVisible.value = true
}

// 删除
const handleDelete = async (row: Equipment) => {
  await ElMessageBox.confirm('确定删除该设备吗？', '提示')
  await deleteEquipment(row.id)
  ElMessage.success('删除成功')
  loadData()
}

// 批量删除
const handleBatchDelete = async () => {
  await ElMessageBox.confirm(`确定删除选中的 ${selectedIds.value.length} 条数据吗？`)
  // await batchDeleteEquipment(selectedIds.value)
  ElMessage.success('删除成功')
  loadData()
}

// 导出
const handleExport = async () => {
  await exportEquipment(queryForm)
  ElMessage.success('导出成功')
}

// 选择变化
const handleSelectionChange = (selection: Equipment[]) => {
  selectedIds.value = selection.map(item => item.id!)
}

// 分页变化
const handleSizeChange = (val: number) => {
  queryForm.pageSize = val
  loadData()
}

const handleCurrentChange = (val: number) => {
  queryForm.pageSize = val
  loadData()
}

// 成功回调
const handleSuccess = () => {
  formVisible.value = false
  loadData()
}

// 状态颜色
const getStatusType = (status: number) => {
  const map: Record<number, string> = {
    0: 'success',
    1: 'warning',
    2: 'danger',
    3: 'info',
    4: ''
  }
  return map[status] || ''
}

onMounted(() => {
  loadData()
})
</script>

```typescript
// frontend/src/api/equipment/index.ts
import request from '@/utils/request'
import type { AxiosPromise } from 'axios'
import type { Equipment, EquipmentQuery, PageResult } from './types'

/**
 * 分页查询设备
 */
export function getEquipmentPage(query: EquipmentQuery): AxiosPromise<PageResult<Equipment>> {
  return request({
    url: '/api/equipment/page',
    method: 'get',
    params: query
  })
}

/**
 * 获取设备详情
 */
export function getEquipmentById(id: number): AxiosPromise<Equipment> {
  return request({
    url: `/api/equipment/${id}`,
    method: 'get'
  })
}

/**
 * 创建设备
 */
export function createEquipment(data: any): AxiosPromise {
  return request({
    url: '/api/equipment',
    method: 'post',
    data
  })
}

/**
 * 更新设备
 */
export function updateEquipment(id: number, data: any): AxiosPromise {
  return request({
    url: `/api/equipment/${id}`,
    method: 'put',
    data
  })
}

/**
 * 删除设备
 */
export function deleteEquipment(id: number): AxiosPromise {
  return request({
    url: `/api/equipment/${id}`,
    method: 'delete'
  })
}

/**
 * 导出设备
 */
export function exportEquipment(query: EquipmentQuery): AxiosPromise {
  return request({
    url: '/api/equipment/export',
    method: 'get',
    params: query,
    responseType: 'blob'
  })
}

## lnngfar frontend 功能需求清单

### 核心能力

| 能力 | 说明 | 实现方式 |
|------|------|----------|
| **页面生成** | 根据模块名生成标准页面结构 | Template Engine |
| **组件生成** | 生成表单、列表等通用组件 | Template Engine |
| **API 生成** | 根据后端 Controller 自动生成 | 代码解析 + 模板 |
| **类型生成** | 根据后端 DTO 生成 TypeScript 类型 | 代码解析 |
| **AI 辅助** | 自然语言描述生成页面 | AI Runtime |
| **布局生成** | 生成标准后台布局 | Template Engine |

### 模板结构

```
