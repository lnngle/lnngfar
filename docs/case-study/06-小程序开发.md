2. 解析设备二维码获取设备ID
3. 调用后端 API 发起借用申请

```vue
<!-- uniapp/src/pages/scan-borrow/index.vue -->
<template>
  <view class="scan-borrow">
    <!-- 扫码按钮 -->

      </view>

        <text class="label">存放位置</text>
        <text class="value">{{ equipmentInfo.location }}</text>
      </view>
    </view>

    <!-- 借用表单 -->
    <view v-if="equipmentInfo && equipmentInfo.status === 0" class="borrow-form">
      <view class="form-item">
        <text class="label">借用用途</text>
        <textarea 
          v-model="form.purpose" 
          placeholder="请输入借用用途"
          class="textarea"
        />
      </view>
      
      <view class="form-item">
        <text class="label">预计归还</text>
        <picker 
          mode="date" 
          :value="form.expectReturn" 
          @change="onDateChange"
        >
          <view class="picker-value">
            {{ form.expectReturn || '请选择日期' }}
          </view>
        </picker>
      </view>
      
      <button class="submit-btn" type="primary" @click="handleSubmit">
        提交借用申请
      </button>
    </view>

    <!-- 结果提示 -->
    <view v-if="resultMsg" class="result-tip" :class="resultType">
      {{ resultMsg }}
    </view>

</template>

<script setup lang="ts">
import { ref, reactive } from 'vue'
import { onLoad } from '@dcloudio/uni-app'
import { getEquipmentByCode, createBorrow } from '@/api/equipment'

// 设备信息
const equipmentInfo = ref<any>(null)

// 表单
const form = reactive({
  purpose: '',
  expectReturn: ''
})

// 结果
const resultMsg = ref('')
const resultType = ref('')

// 扫码
const handleScan = async () => {
  try {
    // 调用扫码
    const scanResult = await new Promise((resolve, reject) => {
      uni.scanCode({
        success: (res) => resolve(res),
        fail: (err) => reject(err)
      })
    })
    
    // 解析二维码
    const code = (scanResult as any).result || (scanResult as any).code
    // 二维码格式：EQUIPMENT:{id}:{code}
    const match = code.match(/^EQUIPMENT:(\d+):(.+)$/)
    if (!match) {
      showResult('无效的设备二维码', 'error')
      return
    }
    
    const [_, id, equipmentCode] = match
    
    // 获取设备信息
    const res = await getEquipmentByCode(equipmentCode)
    equipmentInfo.value = res.data
    
    // 如果是借用状态，自动填写
    if (equipmentInfo.value.status === 0) {
      // 设置默认归还日期为7天后
      const date = new Date()
      date.setDate(date.getDate() + 7)
      form.expectReturn = date.toISOString().split('T')[0]
    }
  } catch (err: any) {
    showResult('扫码失败：' + err.message, 'error')
  }
}

// 提交借用
const handleSubmit = async () => {
  if (!form.purpose.trim()) {
    uni.showToast({ title: '请填写借用用途', icon: 'none' })
    return
  }
  
  if (!form.expectReturn) {
    uni.showToast({ title: '请选择预计归还日期', icon: 'none' })
    return
  }
  
  try {
    uni.showLoading({ title: '提交中...' })
    
    await createBorrow({
      equipmentId: equipmentInfo.value.id,
      purpose: form.purpose,
      expectReturn: form.expectReturn
    })
    
    uni.hideLoading()
    showResult('借用申请提交成功，等待审批', 'success')
    
    // 3秒后返回
    setTimeout(() => {
      uni.navigateBack()
    }, 3000)
  } catch (err: any) {
    uni.hideLoading()
    showResult(err.message || '提交失败', 'error')
  }
}

// 日期选择
const onDateChange = (e: any) => {
  form.expectReturn = e.detail.value
}

// 显示结果
const showResult = (msg: string, type: 'success' | 'error') => {
  resultMsg.value = msg
  resultType.value = type
}

// 状态样式
const getStatusClass = (status: number) => {
  const map: Record<number, string> = {
    0: 'status-idle',
    1: 'status-using',
    2: 'status-repair',
    3: 'status-scrap'
  }
  return map[status] || ''
}

const getStatusText = (status: number) => {
  const map: Record<number, string> = {
    0: '空闲',
    1: '使用中',
    2: '维修中',
    3: '报废'
  }
  return map[status] || '未知'
}
</script>

## API 层生成

```bash
# 根据后端 API 自动生成小程序 API
$ lnngfar uniapp api generate

✓ API 生成完成！

uniapp/src/api/
├── equipment.ts          # 设备 API

├── user.ts              # 用户 API
└── index.ts             # API 入口

```typescript
// uniapp/src/api/borrow.ts
import request from '@/utils/request'
import type { BorrowRecord, BorrowCreateDTO } from '@/types/borrow'

/**
 * 发起借用申请
 */
export function createBorrow(data: BorrowCreateDTO): Promise<void> {
  return request.post('/api/borrow', data)
}

/**
 * 获取我的借用记录
 */
export function getMyBorrowList(params: {
  pageNum?: number
  pageSize?: number
}): Promise<{ data: { records: BorrowRecord[], total: number } }> {
  return request.get('/api/borrow/my', { params })
}

/**
 * 归还设备
 */
export function returnEquipment(id: number, remark?: string): Promise<void> {
  return request.post(`/api/borrow/${id}/return`, { remark })
}
```

## lnngfar uniapp 功能需求清单

### 核心能力

| 能力 | 说明 | 实现方式 |
|------|------|----------|
| **页面生成** | 创建标准 uni-app 页面结构 | Template Engine |
| **组件生成** | 生成 uni-ui 组件 | Template Engine |
| **API 生成** | 根据后端 API 生成调用代码 | 代码解析 |
| **类型生成** | 生成 TypeScript 类型 | 代码解析 |
| **AI 辅助** | 自然语言描述生成页面 | AI Runtime |
| **多端适配** | 支持微信/支付宝/H5 等 | uni-app 内置 |

### 模板结构

```
uniapp-template/
├── src/
│   ├── api/              # API 层模板
